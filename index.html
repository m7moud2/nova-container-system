Installation

One command to install on Linux, macOS, or WSL

bash 
curl --proto '=https' --tlsv1.2 -sSf https://nova.sh/install | sh

Build from Source

bash 
git clone https://github.com/nova-container/core
cd core
cargo build --release
./target/release/nova --version

Quick Start

bash 
# Run a container
nova run app.wasm

# Run with replicas
nova run app.wasm --replicas 10

# With resource limits
nova run app.wasm --fuel 1000000 --memory 512

Advanced Usage

bash 
# Mount volumes
nova run app.wasm --map-dir ./data:/data

# Network between containers
nova run server.wasm --replicas 2

# Full example
nova run app.wasm \
  --replicas 5 \
  --fuel 10000000 \
  --memory 1024 \
  --map-dir ./storage:/app/data
